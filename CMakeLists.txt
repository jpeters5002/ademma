cmake_minimum_required(VERSION 3.13)

if(NOT CMAKE_BUILD_TYPE AND NOT CMAKE_CONFIGURATION_TYPES)
    set(CMAKE_BUILD_TYPE Release CACHE STRING "Choose the type of build." FORCE)
    # set available options for GUI
    set_property(CACHE CMAKE_BUILD_TYPE PROPERTY STRINGS "Debug" "Release" "MinSizeRel" "RelWithDebInfo")
endif()

project(ademma VERSION 0.2.1 LANGUAGES CXX)

set(GENERATED_FILE_WARNING_COMMENT_CPP "// THIS FILE WAS GENERATED BY CMAKE")
configure_file(core_include/cmake_template_version_info.hpp.in ${CMAKE_CURRENT_SOURCE_DIR}/core_include/cmake_generated_version_info.hpp)
set(CORE_SOURCES
	core_src/c_motivic_adem_math.cpp
	core_src/c_motivic_adem_monomial.cpp
	core_src/c_motivic_adem_polynomial.cpp
	core_src/classical_adem_math.cpp
	core_src/classical_adem_monomial.cpp
	core_src/classical_adem_polynomial.cpp
	core_src/general_adem_math.cpp
	core_src/parsing_info.cpp
	core_src/r_motivic_adem_math.cpp
	core_src/r_motivic_adem_monomial.cpp
	core_src/r_motivic_adem_polynomial.cpp
	core_src/steenrod_square.cpp
)
set(TUI_SOURCES
    tui_src/main.cpp
)
add_executable(${PROJECT_NAME} ${TUI_SOURCES} ${CORE_SOURCES})
target_include_directories(${PROJECT_NAME} PUBLIC
	core_include
)

option(BUILD_TESTING "Build the project's tests" OFF)

if (BUILD_TESTING)
    set(TEST_PROJECT_NAME test_${PROJECT_NAME})
    set(TEST_SOURCES
        test_src/main.cpp
        test_src/testing_helper_functions.cpp
        test_src/test_cl_prod_binary_degrees_1_to_n.cpp
        test_src/test_rm_two_factor.cpp
        test_src/test_sq1_sq2n.cpp
        test_src/test_sq2nminus1_sqn.cpp
        test_src/test_sq2tothen_tothe2nplus2.cpp
    )
    add_executable(${TEST_PROJECT_NAME} ${TEST_SOURCES} ${CORE_SOURCES})
    target_include_directories(${TEST_PROJECT_NAME} PUBLIC
        test_include
        core_include
    )
endif()
